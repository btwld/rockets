# @bitwild publish order (dependencies first).
# Prereq: yarn install && yarn build && yarn npm login --publish
#
# All packages:
#   yarn publish:bitwild
#
# Per package (from repo root):
#   yarn workspace <name> npm publish --access public --tag alpha
#
# From package dir:
#   yarn npm publish --access public --tag alpha
#
# Yarn 4 rewrites workspace:^ to ^<version> in the packed manifest.
